## Usage

```tsx
import { StepIndicator } from "@/components/ui/step-indicator"

const steps = [
  { id: "1", label: "Account", status: "complete" },
  { id: "2", label: "Profile", status: "active" },
  { id: "3", label: "Settings", status: "pending" }
]

export default function Example() {
  return <StepIndicator steps={steps} />
}
```

## Examples

### Horizontal Variant

The default horizontal variant displays steps in a row with connecting lines.

```tsx
<StepIndicator steps={steps} />
```

### Vertical Variant

Use the vertical variant for forms or processes that benefit from a top-to-bottom flow.

```tsx
<StepIndicator steps={steps} variant="vertical" />
```

### With Descriptions

Add descriptions to provide more context for each step.

```tsx
const stepsWithDescriptions = [
  { 
    id: "1", 
    label: "Create Account", 
    description: "Enter your email and password",
    status: "complete" 
  },
  { 
    id: "2", 
    label: "Verify Email", 
    description: "Check your inbox for verification",
    status: "active" 
  },
  { 
    id: "3", 
    label: "Complete Profile", 
    description: "Add your personal information",
    status: "pending" 
  }
]

<StepIndicator steps={stepsWithDescriptions} variant="vertical" />
```

## API Reference

### StepIndicator

| Prop | Type | Default | Description |
| --- | --- | --- | --- |
| steps | `Step[]` | - | Array of step objects |
| className | `string` | `""` | Additional CSS classes |
| variant | `"horizontal" \| "vertical"` | `"horizontal"` | Display orientation |

### Step Interface

```typescript
interface Step {
  id: string
  label: string
  description?: string
  status: "pending" | "active" | "complete"
}
```
